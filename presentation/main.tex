\begin{frame}{Title and Objective}
    \vspace{0.4em}

    {\large \projectsubtitle}

    \vspace{1.2em}
    \begin{block}{Objective}
        Evaluate whether a feed-forward neural network can learn the Black-Scholes European call pricing function and act as an efficient pricing surrogate.
    \end{block}

    \vfill
    \small Project group \hfill Machine Learning for Finance
\end{frame}

\begin{frame}{Research Question}
    \begin{block}{Research question}
        Can a feed-forward neural network accurately approximate the Black-Scholes pricing function for European call options?
    \end{block}

    \vspace{0.5em}
    The project compares:
    \begin{itemize}
        \item the analytical Black-Scholes formula;
        \item Monte Carlo simulation;
        \item a supervised neural network pricing surrogate.
    \end{itemize}
\end{frame}

\begin{frame}{Black-Scholes SDE}
    The underlying asset follows the geometric Brownian motion
    \[
        dS_t = r S_t\,dt + \sigma S_t\,dW_t,
    \]
    where \(r\) is the risk-free rate, \(\sigma\) is volatility and \(W_t\) is a Brownian motion.

    \vspace{0.7em}
    The analytical European call price is used as the ground truth:
    \[
        S_T = S_0 \exp\left((r-\tfrac{1}{2}\sigma^2)T + \sigma W_T\right).
    \]
\end{frame}

\begin{frame}{European Call Option Pricing}
    A European call option has payoff
    \[
        \Phi(S_T) = \max(S_T-K,0).
    \]

    Under the Black-Scholes model, its analytical price is
    \[
        C_{\mathrm{BS}} = S_0 N(d_1) - K e^{-rT}N(d_2),
    \]
    with
    \[
        d_1 = \frac{\log(S_0/K)+(r+\tfrac{1}{2}\sigma^2)T}{\sigma\sqrt{T}},
        \qquad
        d_2 = d_1-\sigma\sqrt{T}.
    \]
\end{frame}

\begin{frame}{Synthetic Dataset}
    Synthetic observations are sampled over the input space:
    \[
        x = (S_0, K, T, r, \sigma),
        \qquad
        y = C_{\mathrm{BS}}(S_0,K,T,r,\sigma).
    \]

    \vspace{0.5em}
    \begin{center}
        \begin{tabular}{ll}
            \toprule
            Variable              & Range          \\
            \midrule
            Initial price \(S_0\) & \([50, 150]\)  \\
            Strike \(K\)          & \([50, 150]\)  \\
            Maturity \(T\)        & \([0.1, 2]\)   \\
            Risk-free rate \(r\)  & \([0, 0.05]\)  \\
            Volatility \(\sigma\) & \([0.1, 0.6]\) \\
            \bottomrule
        \end{tabular}
    \end{center}
\end{frame}

\begin{frame}{Neural Network as Pricing Surrogate}
    The neural network learns the pricing map
    \[
        f_\theta: (S_0, K, T, r, \sigma) \mapsto C_{\mathrm{BS}}.
    \]

    \vspace{0.5em}
    Final architecture:
    \begin{itemize}
        \item fully connected feed-forward network;
        \item moneyness \(S_0/K\) included as engineered feature;
        \item SiLU hidden activations;
        \item mean squared error loss;
        \item Adam optimizer;
        \item train/validation/test split with feature scaling.
    \end{itemize}
\end{frame}

\begin{frame}{Experimental Pipeline}
    \begin{enumerate}
        \item Sample synthetic contracts.
        \item Compute Black-Scholes analytical prices.
        \item Train the neural network on the supervised pricing task.
        \item Estimate selected prices with Monte Carlo simulation.
        \item Compare errors and analyse where approximation is harder.
    \end{enumerate}
\end{frame}

\begin{frame}{Evaluation Metrics}
    The main quantitative metrics are:
    \begin{itemize}
        \item Mean Absolute Error;
        \item Root Mean Squared Error;
        \item coefficient of determination \(R^2\);
        \item relative error diagnostics where meaningful.
    \end{itemize}

    \vspace{0.5em}
    Error analysis is performed against moneyness \(S_0/K\), maturity \(T\), and volatility \(\sigma\).
\end{frame}

\begin{frame}{Main Results}
    Final experiment:
    \begin{itemize}
        \item 100,000 synthetic option contracts;
        \item neural network trained up to 200 epochs with early stopping;
        \item Monte Carlo benchmark with 50,000 paths on 512 test options.
    \end{itemize}

\end{frame}

\begin{frame}{Monte Carlo Benchmark}
    Simulated terminal prices are generated as
    \[
        S_T = S_0 \exp\left((r-\tfrac{1}{2}\sigma^2)T + \sigma\sqrt{T}Z\right),
        \qquad Z \sim \mathcal{N}(0,1).
    \]

    The discounted estimator is
    \[
        \widehat{C}_{\mathrm{MC}}
        = e^{-rT}\frac{1}{M}\sum_{m=1}^{M}\max(S_T^{(m)}-K,0).
    \]

    \vspace{0.4em}
    \begin{itemize}
        \item Evaluated on 512 deterministic test options.
        \item Provides a classical numerical benchmark with sampling error.
        \item The trained neural network gives deterministic forward-pass predictions.
    \end{itemize}
\end{frame}

\begin{frame}{Runtime Benchmark}
    \begin{center}

            \toprule

    \vspace{0.4em}
    \begin{block}{Surrogate pricing point}
        After training, neural inference was about \(40\times\) faster than Monte Carlo for pricing 1,000 options in the benchmark.
    \end{block}
\end{frame}

\begin{frame}{Discussion}
    \begin{itemize}
        \item The network accurately learns the Black-Scholes pricing map in the sampled domain.
        \item Monte Carlo is also accurate, but remains simulation-based and stochastic.
        \item The neural network is best interpreted as a pricing surrogate after training.
        \item The experiment validates the controlled Black-Scholes setting before moving to richer dynamics.
    \end{itemize}
\end{frame}

\begin{frame}{Limitations}
    \begin{itemize}
        \item The first version uses synthetic data generated under Black-Scholes assumptions.
        \item The model prices European calls only.
        \item The target function is known analytically, so the neural network is evaluated as a surrogate rather than as a replacement for financial theory.
        \item Generalization outside the sampled parameter ranges is not guaranteed.
    \end{itemize}
\end{frame}

\begin{frame}{Conclusions and Future Work}
    The research question is answered positively in the controlled Black-Scholes setting.

    \vspace{0.5em}
    \begin{block}{Main takeaway}
        A feed-forward neural network can approximate European call prices generated by the Black-Scholes formula with very high test-set accuracy.
    \end{block}

    \vspace{0.4em}
    Future extensions may include stochastic volatility, Heston dynamics, American options and Greeks estimation.
\end{frame}
